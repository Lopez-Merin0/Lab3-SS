Documentación de la API REST

Expandir ↓
POST /api/upload Recibe archivo y lo guarda temporalmente
// Request (multipart/form-data) Content-Type: multipart/form-data Body: file = <archivo_binario> // Response 201 { "fileId": "file_abc123", "filename": "documento.pdf", "size": 204800, "status": "uploaded" }
POST /api/scan/:fileId Escanear archivo con ClamAV
// Response 202 { "scanId": "scan_abc123", "fileId": "file_abc123", "status": "pending" }
GET /api/status/:scanId Consulta estado del escaneo
{ "scanId": "scan_abc123", "status": "pending" | "scanning" | "done" | "error" }
GET /api/result/:scanId Obtiene resultado final
{ "scanId": "scan_abc123", "status": "clean" | "infected", "threats": ["Win.Trojan.Agent-1234"], "action": "quarantined" }
DELETE /api/cleanup/:fileId Limpia archivos temporales
{ "cleaned": true, "fileId": "file_abc123" }
